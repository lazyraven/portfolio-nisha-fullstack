Side Effects ⭐ Interview Fav
useEffect
  • ▸Runs after render — sync with externals
  • ▸Cleanup functions — prevent memory leaks
  • ▸Dependency array — controls when effect runs
  • ▸[] = run once, [val] = on val change
  • ▸No array = runs on every render
🎯
Interview Q: When does useEffect run?After paint. Dep array controls frequency. Cleanup runs before next effect + unmount.

Interview-Level Code
💡 Todo App — What Interviews Test
TodoApp.jsx — tests: useState, useEffect, controlled input 
import { useState, useEffect } from "react";

export default function TodoApp() {
  const [todos, setTodos] = useState(() => {
    return JSON.parse(localStorage.getItem("todos")) || [];
  }); // lazy initializer — runs once

  const [input, setInput] = useState("");

  useEffect(() => {
    localStorage.setItem("todos", JSON.stringify(todos));
  }, [todos]); // runs whenever todos changes

  const addTodo = () => {
    if (!input.trim()) return;
    setTodos([...todos, { id: Date.now(), text: input, done: false }]);
    setInput("");
  };

  const toggleTodo = (id) =>
    setTodos(todos.map(t => t.id === id ? { ...t, done: !t.done } : t));

  return (
    <div>
      <input value={input} onChange={(e) => setInput(e.target.value)} />
      <button onClick={addTodo}>Add</button>
      {todos.map(todo => (
        <div key={todo.id} onClick={() => toggleTodo(todo.id)}>
          {todo.text} {todo.done ? "✔" : ""}
        </div>
      ))}
    </div>
  );
}
