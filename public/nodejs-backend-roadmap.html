server.js — Express REST API example
Express.js REST 
const express = require('express');
const app = express();
app.use(express.json());   // parse JSON bodies

let todos = [];

// GET all todos
app.get('/api/todos', (req, res) => {
  res.json({ success: true, data: todos });
});

// POST create todo
app.post('/api/todos', (req, res) => {
  const { title } = req.body;
  if (!title) return res.status(400).json({ error: 'Title required' });
  const todo = { id: Date.now(), title, done: false };
  todos.push(todo);
  res.status(201).json({ success: true, data: todo });
});

// Centralized error handler (4 params = error middleware)
app.use((err, req, res, next) => {
  res.status(500).json({ error: err.message });
});

app.listen(3000, () => console.log('🚀 Server running on :3000'));

Watch Out
Common Mistakes to Avoid
❌ Skipping Event Loop
The event loop is Node.js's superpower. Skipping it means you'll never understand why async code behaves the way it does.
❌ Tutorial Hell
Only watching tutorials without writing code. After any tutorial, build a variation from scratch immediately.
❌ Not Building Projects
Projects reveal gaps tutorials hide. Every job interview will ask about real things you built, not videos you watched.
❌ Ignoring Error Handling
Unhandled promise rejections crash Node. Centralized error handling is a must in every production app.
